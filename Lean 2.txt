import Mathlib.Analysis.SpecialFunctions.Sqrt
import Mathlib.Data.Real.Basic
import Mathlib.Tactic

open Real

noncomputable def discreteLMC (a_p p : ℝ) : ℝ := a_p / (2 * sqrt p)
def lmcSigma (σ₀ β t : ℝ) : ℝ := σ₀ * (1 + β * t)
noncomputable def ellipticityRatio (β : ℝ) : ℝ := (1 + β) / (1 - β)

-- ============================================
-- ORIGINAL 6 THEOREME
-- ============================================

theorem discreteLMC_bounded (a_p p : ℝ) 
    (hp : 0 < p) (h_hasse : |a_p| ≤ 2 * sqrt p) :
    |discreteLMC a_p p| ≤ 1 := by
  unfold discreteLMC
  rw [abs_div, abs_mul, abs_of_pos (by linarith : (0:ℝ) < 2)]
  rw [abs_of_nonneg (sqrt_nonneg p)]
  have h1 : 0 < 2 * sqrt p := by
    apply mul_pos; linarith; exact sqrt_pos.mpr hp
  rw [div_le_one h1]; exact h_hasse

theorem uniform_ellipticity_lower (σ₀ β t : ℝ)
    (hσ : 0 < σ₀) (hβ_nn : 0 ≤ β) (ht : -1 ≤ t) :
    σ₀ * (1 - β) ≤ lmcSigma σ₀ β t := by
  unfold lmcSigma
  have h1 : 1 - β ≤ 1 + β * t := by
    have : -β ≤ β * t := by
      calc -β = -(β * 1) := by ring
           _ = β * (-1) := by ring
           _ ≤ β * t := mul_le_mul_of_nonneg_left ht hβ_nn
    linarith
  exact mul_le_mul_of_nonneg_left h1 (le_of_lt hσ)

theorem uniform_ellipticity_upper (σ₀ β t : ℝ)
    (hσ : 0 < σ₀) (hβ_nn : 0 ≤ β) (ht : t ≤ 1) :
    lmcSigma σ₀ β t ≤ σ₀ * (1 + β) := by
  unfold lmcSigma
  have h1 : 1 + β * t ≤ 1 + β := by
    have : β * t ≤ β := by
      calc β * t ≤ β * 1 := mul_le_mul_of_nonneg_left ht hβ_nn
           _ = β := by ring
    linarith
  exact mul_le_mul_of_nonneg_left h1 (le_of_lt hσ)

theorem conductivity_positive (σ₀ β t : ℝ)
    (hσ : 0 < σ₀) (hβ_nn : 0 ≤ β) (hβ_lt : β < 1) (ht : |t| ≤ 1) :
    0 < lmcSigma σ₀ β t := by
  unfold lmcSigma
  have ht_lower : -1 ≤ t := by rw [abs_le] at ht; exact ht.1
  have h1 : 0 < 1 + β * t := by
    have : -β ≤ β * t := by
      calc -β = β * (-1) := by ring
           _ ≤ β * t := mul_le_mul_of_nonneg_left ht_lower hβ_nn
    linarith
  exact mul_pos hσ h1

theorem ellipticityRatio_ge_one (β : ℝ)
    (hβ_nn : 0 ≤ β) (hβ_lt : β < 1) :
    1 ≤ ellipticityRatio β := by
  unfold ellipticityRatio
  have h_denom : 0 < 1 - β := by linarith
  have : 1 - β ≤ 1 + β := by linarith
  calc 1 = (1 - β) / (1 - β) := by rw [div_self]; linarith
       _ ≤ (1 + β) / (1 - β) := div_le_div_of_nonneg_right this (le_of_lt h_denom)

theorem lmc_main_theorem (σ₀ β t : ℝ)
    (hσ : 0 < σ₀) (hβ_nn : 0 ≤ β) (hβ_lt : β < 1) (ht : |t| ≤ 1) :
    (σ₀ * (1 - β) ≤ lmcSigma σ₀ β t ∧ lmcSigma σ₀ β t ≤ σ₀ * (1 + β)) ∧
    (0 < lmcSigma σ₀ β t) ∧ (1 ≤ ellipticityRatio β) := by
  constructor
  · constructor
    · have ht_lower : -1 ≤ t := by rw [abs_le] at ht; exact ht.1
      exact uniform_ellipticity_lower σ₀ β t hσ hβ_nn ht_lower
    · have ht_upper : t ≤ 1 := by rw [abs_le] at ht; exact ht.2
      exact uniform_ellipticity_upper σ₀ β t hσ hβ_nn ht_upper
  constructor
  · exact conductivity_positive σ₀ β t hσ hβ_nn hβ_lt ht
  · exact ellipticityRatio_ge_one β hβ_nn hβ_lt

-- ============================================
-- NEU A1: MONOTONIE VON σ IN t
-- ============================================

theorem lmcSigma_monotone_in_t (σ₀ β t₁ t₂ : ℝ)
    (hσ : 0 < σ₀) (hβ : 0 < β) (ht : t₁ ≤ t₂) :
    lmcSigma σ₀ β t₁ ≤ lmcSigma σ₀ β t₂ := by
  unfold lmcSigma
  have h1 : 1 + β * t₁ ≤ 1 + β * t₂ := by
    have : β * t₁ ≤ β * t₂ := mul_le_mul_of_nonneg_left ht (le_of_lt hβ)
    linarith
  exact mul_le_mul_of_nonneg_left h1 (le_of_lt hσ)

-- ============================================
-- NEU A2: ELLIPTICITY RATIO MONOTONIE
-- ============================================

theorem ellipticityRatio_monotone (β₁ β₂ : ℝ)
    (h1 : 0 ≤ β₁) (h2 : β₁ ≤ β₂) (h3 : β₂ < 1) :
    ellipticityRatio β₁ ≤ ellipticityRatio β₂ := by
  unfold ellipticityRatio
  have h_denom1 : 0 < 1 - β₁ := by linarith
  have h_denom2 : 0 < 1 - β₂ := by linarith
  have h_num : 1 + β₁ ≤ 1 + β₂ := by linarith
  have h_denom_rev : 1 - β₂ ≤ 1 - β₁ := by linarith
  calc (1 + β₁) / (1 - β₁)
      ≤ (1 + β₂) / (1 - β₁) := div_le_div_of_nonneg_right h_num (le_of_lt h_denom1)
    _ ≤ (1 + β₂) / (1 - β₂) := by {
        apply div_le_div_of_nonneg_left
        · linarith
        · exact h_denom2
        · exact h_denom_rev
      }

-- ============================================
-- NEU A3: STETIGKEIT (strukturell)
-- ============================================

theorem lmcSigma_continuous_structure :
    ∀ σ₀ β t, lmcSigma σ₀ β t = σ₀ + σ₀ * β * t := by
  intros; unfold lmcSigma; ring

theorem ellipticityRatio_structure (β : ℝ) :
    ellipticityRatio β = (1 + β) / (1 - β) := rfl

-- ============================================
-- NEU A4: BEISPIELE
-- ============================================

example : discreteLMC (-4) 5 = -4 / (2 * sqrt 5) := rfl

theorem example_sigma_p5 :
    lmcSigma 1 (1/2) (-2 / sqrt 5) = 1 - 1 / sqrt 5 := by
  unfold lmcSigma; ring

-- ============================================
-- NEU A5: FLUX BOUND (einfache Version)
-- ============================================

theorem flux_simple_bound (σ₀ β t : ℝ)
    (hσ : 0 < σ₀) (hβ_nn : 0 ≤ β) (ht : |t| ≤ 1) :
    lmcSigma σ₀ β t ≤ σ₀ * (1 + β) := by
  have ht_upper : t ≤ 1 := by rw [abs_le] at ht; exact ht.2
  exact uniform_ellipticity_upper σ₀ β t hσ hβ_nn ht_upper
