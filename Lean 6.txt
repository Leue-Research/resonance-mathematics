-- ════════════════════════════════════════════════════════════════════════
-- 1. KONSTANTEN
-- ════════════════════════════════════════════════════════════════════════

/-- Vakuum-Spektral-Eichkonstante Δ∞ = 6.5307 -/
noncomputable def DeltaInf : ℝ := 6.5307

/-- Drift-Kopplungskonstante α = 0.0683 -/
noncomputable def Alpha : ℝ := 0.0683

/-- Vakuum-Schwelle T₀ = 1.451369 -/
noncomputable def T0 : ℝ := 1.451369

-- ════════════════════════════════════════════════════════════════════════
-- 2. LAMBERT-W FUNKTION (axiomatisch)
--    Lambert-W existiert nicht in Mathlib. Wir führen sie als opaque
--    Funktion mit ihrer definierenden Eigenschaft ein.
--    Mathematisch: "Sei W die Hauptzweig-Lösung von W(x)·e^{W(x)} = x."
-- ════════════════════════════════════════════════════════════════════════

/-- Lambert-W Funktion (Hauptzweig) -/
noncomputable opaque lambertW : ℝ → ℝ

/-- Axiom: W(x)·e^{W(x)} = x für x > 0 -/
axiom lambertW_spec (x : ℝ) (hx : 0 < x) :
  lambertW x * exp (lambertW x) = x

/-- Axiom: W(x) > 0 für x > 0 -/
axiom lambertW_pos (x : ℝ) (hx : 0 < x) : 0 < lambertW x

/-- Axiom: W ist monoton steigend für x > 0 -/
axiom lambertW_mono {x y : ℝ} (hx : 0 < x) (hxy : x ≤ y) :
  lambertW x ≤ lambertW y

-- ════════════════════════════════════════════════════════════════════════
-- 3. LOGARITHMISCHES INTEGRAL (axiomatisch)
--    li(t) = ∫₂ᵗ 1/ln(x) dx existiert nicht als benannte Funktion
--    in Mathlib. Wir geben nur die benötigten Eigenschaften an.
-- ════════════════════════════════════════════════════════════════════════

/-- Logarithmisches Integral li(t) -/
noncomputable opaque logIntegral : ℝ → ℝ

/-- Axiom: li(t) > 0 für t > 2 -/
axiom logIntegral_pos (t : ℝ) (ht : 2 < t) : 0 < logIntegral t

/-- Axiom: li ist streng monoton steigend für t > 2 -/
axiom logIntegral_mono {s t : ℝ} (hs : 2 < s) (hst : s ≤ t) :
  logIntegral s ≤ logIntegral t

-- ════════════════════════════════════════════════════════════════════════
-- 4. PRIME TIME: tₙ = Σ ln(pᵢ)
-- ════════════════════════════════════════════════════════════════════════

/-- Prime Time: Summe der Logarithmen der Primzahlen -/
noncomputable def primeTime (primes : List ℕ) : ℝ :=
  (primes.map (fun p => Real.log (p : ℝ))).sum

theorem primeTime_nil : primeTime [] = 0 := by
  unfold primeTime; simp

theorem primeTime_singleton (p : ℕ) :
    primeTime [p] = Real.log (p : ℝ) := by
  unfold primeTime; simp

theorem primeTime_append (xs ys : List ℕ) :
    primeTime (xs ++ ys) = primeTime xs + primeTime ys := by
  unfold primeTime; simp [List.map_append, List.sum_append]

-- ════════════════════════════════════════════════════════════════════════
-- 5. DRIFT: dₖ = (pₖ - pₖ₋₁)/ln(pₖ₋₁) - 1
-- ════════════════════════════════════════════════════════════════════════

/-- Einzelner Drift zwischen konsekutiven Primzahlen -/
noncomputable def singleDrift (p_prev p_curr : ℕ) : ℝ :=
  ((p_curr : ℝ) - (p_prev : ℝ)) / Real.log (p_prev : ℝ) - 1

/-- Kumulativer Drift über eine Primzahlliste -/
noncomputable def cumulativeDrift : List ℕ → ℝ
  | [] => 0
  | [_] => 0
  | p₁ :: p₂ :: rest =>
    singleDrift p₁ p₂ + cumulativeDrift (p₂ :: rest)

theorem cumulativeDrift_nil : cumulativeDrift [] = 0 := rfl
theorem cumulativeDrift_singleton (p : ℕ) : cumulativeDrift [p] = 0 := rfl

-- ════════════════════════════════════════════════════════════════════════
-- 6. LEUE MAP: Φ(t) = 2π · li(t) / W(li(t)/e)
-- ════════════════════════════════════════════════════════════════════════

/-- Leue Map: Die zentrale Transformation von Prime Time in Spektralraum -/
noncomputable def leueMap (t : ℝ) : ℝ :=
  2 * π * logIntegral t / lambertW (logIntegral t / exp 1)

/-- Der Nenner der Leue Map ist nicht-null für t > 2 -/
theorem leueMap_denom_ne_zero (t : ℝ) (ht : 2 < t) :
    lambertW (logIntegral t / exp 1) ≠ 0 := by
  have hli : 0 < logIntegral t := logIntegral_pos t ht
  have harg : 0 < logIntegral t / exp 1 := div_pos hli (exp_pos 1)
  exact ne_of_gt (lambertW_pos _ harg)

/-- Leue Map ist positiv für t > 2 -/
theorem leueMap_pos (t : ℝ) (ht : 2 < t) : 0 < leueMap t := by
  unfold leueMap
  have hli : 0 < logIntegral t := logIntegral_pos t ht
  have harg : 0 < logIntegral t / exp 1 := div_pos hli (exp_pos 1)
  have hw : 0 < lambertW (logIntegral t / exp 1) := lambertW_pos _ harg
  apply div_pos
  · apply mul_pos
    · have : (0 : ℝ) < π := pi_pos
      linarith
    · exact hli
  · exact hw

-- ════════════════════════════════════════════════════════════════════════
-- 7. PTQ MASTER-FORMEL: γₙ = Φ(tₙ) - Δ∞ - α·Dₙ
-- ════════════════════════════════════════════════════════════════════════

/-- PTQ-Prädiktor -/
noncomputable def ptqPredict (t_n D_n : ℝ) : ℝ :=
  leueMap t_n - DeltaInf - Alpha * D_n

/-- PTQ-Vorhersagefehler -/
noncomputable def ptqError (t_n D_n gamma_true : ℝ) : ℝ :=
  |ptqPredict t_n D_n - gamma_true|

/-- Master-Formel: Strukturelle Zerlegung -/
theorem ptq_decomposition (t_n D_n : ℝ) :
    ptqPredict t_n D_n = leueMap t_n - DeltaInf - Alpha * D_n := rfl

/-- PTQ ist affin-linear im Drift:
    Die Differenz zweier Vorhersagen hängt nur von der Drift-Differenz ab. -/
theorem ptq_affine_in_drift (t_n D₁ D₂ : ℝ) :
    ptqPredict t_n D₁ - ptqPredict t_n D₂ = Alpha * (D₂ - D₁) := by
  unfold ptqPredict Alpha; ring

/-- Drift-Sensitivität: Wie der Prädiktor auf Drift-Änderung reagiert -/
theorem ptq_drift_sensitivity (t_n D_n δ : ℝ) :
    ptqPredict t_n (D_n + δ) = ptqPredict t_n D_n - Alpha * δ := by
  unfold ptqPredict; ring

/-- Bei Drift = 0 reduziert sich PTQ auf Φ(t) - Δ∞ -/
theorem ptq_zero_drift (t_n : ℝ) :
    ptqPredict t_n 0 = leueMap t_n - DeltaInf := by
  unfold ptqPredict Alpha; ring

/-- Skalierung: Verdopplung des Drifts verschiebt den Prädiktor um α·D -/
theorem ptq_drift_doubling (t_n D_n : ℝ) :
    ptqPredict t_n (2 * D_n) = ptqPredict t_n D_n - Alpha * D_n := by
  unfold ptqPredict; ring

-- ════════════════════════════════════════════════════════════════════════
-- 8. FEHLER-EIGENSCHAFTEN
-- ════════════════════════════════════════════════════════════════════════

/-- Fehler ist nicht-negativ -/
theorem ptqError_nonneg (t_n D_n gamma_true : ℝ) :
    0 ≤ ptqError t_n D_n gamma_true :=
  abs_nonneg _


/-- Fehler ist symmetrisch im wahren Wert -/
theorem ptqError_symm (t_n D_n γ : ℝ) :
    ptqError t_n D_n γ = |γ - ptqPredict t_n D_n| := by
  unfold ptqError; rw [abs_sub_comm]

-- ════════════════════════════════════════════════════════════════════════
-- 9. GAP RULE (Hexa-Primal System)
-- ════════════════════════════════════════════════════════════════════════

/-- Erlaubte Lücken im Hexa-Primal System -/
def isAllowedGap (g : ℕ) : Prop := g = 2 ∨ g = 4 ∨ g = 6

/-- Gap-Kodierung: gerade → +1, ungerade → -1 -/
def gapSign (g : ℕ) : ℤ := if g % 2 = 0 then 1 else -1

/-- Alle erlaubten Lücken sind gerade -/
theorem allowedGap_even (g : ℕ) (h : isAllowedGap g) : g % 2 = 0 := by
  rcases h with rfl | rfl | rfl <;> decide

/-- Erlaubte Lücken haben stets positives Vorzeichen -/
theorem allowedGap_sign_pos (g : ℕ) (h : isAllowedGap g) : gapSign g = 1 := by
  unfold gapSign; simp [allowedGap_even g h]

/-- Alle erlaubten Lücken sind positiv -/
theorem allowedGap_pos (g : ℕ) (h : isAllowedGap g) : 0 < g := by
  rcases h with rfl | rfl | rfl <;> omega

/-- Alle erlaubten Lücken sind ≤ 6 -/
theorem allowedGap_le_six (g : ℕ) (h : isAllowedGap g) : g ≤ 6 := by
  rcases h with rfl | rfl | rfl <;> omega

/-- 2 ist eine erlaubte Lücke -/
theorem allowedGap_two : isAllowedGap 2 := Or.inl rfl

/-- 4 ist eine erlaubte Lücke -/
theorem allowedGap_four : isAllowedGap 4 := Or.inr (Or.inl rfl)

/-- 6 ist eine erlaubte Lücke -/
theorem allowedGap_six : isAllowedGap 6 := Or.inr (Or.inr rfl)

-- ════════════════════════════════════════════════════════════════════════
-- 10. DRIFT-STABILITÄT
-- ════════════════════════════════════════════════════════════════════════

/-- Bounded Drift: ∃ C > 0, ∀ n, |Dₙ| ≤ C·n -/
def HasBoundedDrift (drift_seq : ℕ → ℝ) : Prop :=
  ∃ C : ℝ, 0 < C ∧ ∀ n : ℕ, |drift_seq n| ≤ C * n

/-- Drift-Beitrag zum Fehler ist ebenfalls beschränkt -/
theorem drift_contribution_bounded
    (drift_seq : ℕ → ℝ) (h : HasBoundedDrift drift_seq) :
    ∃ C : ℝ, 0 < C ∧ ∀ n : ℕ, |Alpha * drift_seq n| ≤ C * n := by
  rcases h with ⟨C, hC, hbound⟩
  have hα : (0 : ℝ) < Alpha := by unfold Alpha; norm_num
  refine ⟨Alpha * C, mul_pos hα hC, ?_⟩
  intro n
  have h1 : |Alpha * drift_seq n| = Alpha * |drift_seq n| := by
    rw [abs_mul, abs_of_pos hα]
  rw [h1]
  calc Alpha * |drift_seq n|
      ≤ Alpha * (C * ↑n) := mul_le_mul_of_nonneg_left (hbound n) (le_of_lt hα)
    _ = Alpha * C * ↑n := by ring

/-- Sup-beschränkter Drift (stärker: |Dₙ| ≤ C für alle n) -/
def HasSupBoundedDrift (drift_seq : ℕ → ℝ) : Prop :=
  ∃ C : ℝ, 0 ≤ C ∧ ∀ n : ℕ, |drift_seq n| ≤ C

/-- Sup-beschränkter Drift ⟹ Drift-Beitrag ist sup-beschränkt -/
theorem sup_drift_contribution_bounded
    (drift_seq : ℕ → ℝ) (h : HasSupBoundedDrift drift_seq) :
    HasSupBoundedDrift (fun n => Alpha * drift_seq n) := by
  rcases h with ⟨C, hC, hbound⟩
  have hα : (0 : ℝ) ≤ Alpha := by unfold Alpha; norm_num
  refine ⟨Alpha * C, mul_nonneg hα hC, ?_⟩
  intro n
  have h1 : |Alpha * drift_seq n| = Alpha * |drift_seq n| := by
    rw [abs_mul, abs_of_nonneg hα]
  rw [h1]
  exact mul_le_mul_of_nonneg_left (hbound n) hα
